const express = require("express");
const http = require("http");
const fs = require("fs");
const axios = require("axios");
const { Server } = require("socket.io");

const app = express();
const server = http.createServer(app);
const io = new Server(server);

const DATA = "./ranking.json";

app.use(express.static("public"));

function loadRanking() {
    return JSON.parse(fs.readFileSync(DATA));
}
function saveRanking(data) {
    fs.writeFileSync(DATA, JSON.stringify(data, null, 2));
}

io.on("connection", socket => {
    socket.emit("ranking", loadRanking());

    socket.on("register", data => {
        io.emit("notify_discord", data);
    });
});

server.listen(3000, () => {
    console.log("✅ Web running http://localhost:3000");
});

module.exports = { loadRanking, saveRanking, io };
